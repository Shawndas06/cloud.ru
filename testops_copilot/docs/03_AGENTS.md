```python
def _find_exact_duplicates(self, tests: List[Dict]) -> List[Dict]:
    """Level 1: Exact Match - —Å—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø–æ —Ö–µ—à—É"""
    duplicates = []
    seen_hashes = {}
    
    for test in tests:
        code_hash = hashlib.sha256(test["test_code"].encode()).hexdigest()
        
        if code_hash in seen_hashes:

                "similarity_score": 1.0
            })
        else:
            seen_hashes[code_hash] = test["test_id"]
    
    return duplicates
```

```python
async def _find_semantic_duplicates(
    self,
    tests: List[Dict],
    threshold: float
) -> List[Dict]:
    """Level 2: Semantic Similarity —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º embeddings"""
    
    # –ì–µ–Ω–µ—Ä–∞—Ü–∏—è embeddings –¥–ª—è –≤—Å–µ—Ö —Ç–µ—Å—Ç–æ–≤
    embeddings = []
    for test in tests:
        test_text = f"{test.get('test_name', '')} {test.get('test_code', '')}"
        embedding = await llm_client.generate_embeddings(test_text)
        embeddings.append(embedding)
    
    # –ü–æ–∏—Å–∫ —Å–µ–º–∞–Ω—Ç–∏—á–µ—Å–∫–∏ –ø–æ—Ö–æ–∂–∏—Ö —Ç–µ—Å—Ç–æ–≤

        for j in range(i + 1, len(tests)):
            similarity = self._cosine_similarity(
                embeddings[i],
                embeddings[j]
            )

                duplicates.append({
                    "test_ids": [tests[i]["test_id"], tests[j]["test_id"]],
                    "type": "semantic",
                    "similarity_score": float(similarity)
                })
```

```python
def _cosine_similarity(self, vec1: np.ndarray, vec2: np.ndarray) -> float:
    """–í—ã—á–∏—Å–ª–µ–Ω–∏–µ –∫–æ—Å–∏–Ω—É—Å–Ω–æ–≥–æ —Å—Ö–æ–¥—Å—Ç–≤–∞"""
    dot_product = np.dot(vec1, vec2)
    norm1 = np.linalg.norm(vec1)
    norm2 = np.linalg.norm(vec2)
    
    if norm1 == 0 or norm2 == 0:
        return 0.0
    
    similarity = dot_product / (norm1 * norm2)
    return float(similarity)
```
